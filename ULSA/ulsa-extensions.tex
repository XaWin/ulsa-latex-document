\newcommand{\UlsaPdfNewSection}[3][-]{%
    \addtocounter{section}{1}%
    \addcontentsline{toc}{section}{\protect\numberline{\thesection}#3}%
    \markboth{}{\thesection.~#3}
    \includepdf[frame,pages={#1},width=\textwidth,pagecommand={}]{#2}%
}

\newcommand{\UlsaPdfNewSubsection}[3][-]{%
    \addtocounter{subsection}{1}%
    \addcontentsline{toc}{subsection}{\protect\numberline{\thesubsection}#3}%
    \markboth{}{\thesection.\thesubsection~#3}
    \includepdf[frame,pages={#1},width=\textwidth,pagecommand={}]{#2}%
}
